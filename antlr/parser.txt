public class Parser {

        this.token = tokens.get(0);

    }

    public Object proximo_token(){
        boolean tem_tokens = this.tokens.size() > 0;
        if (tem_tokens){
            Object current_token = this.tokens.get(0);
            this.tokens.remove(0);
            return current_token;
        }
        return new Object();
    }

    public void entrada_reconhecida(){
        System.out.println("Entrada reconhecida");
    }

    }

    public void executa(){
        this.token = this.proximo_token();

}
